172void Solver::scaleData() {
173 if(not Parameters::isScalingOverridden())
174 return;
175
176 std::string target = Parameters::getScalingTarget();
177 if(target == "objective")
178 problem->scaleObjective(Parameters::getScaling());
179 else if(target.substr(0, 3) == "res" and target.length() >= 4){
180 int index = target[3] - '0';
181 if(index < problem->getNumRes())
182 problem->scaleResource(index, Parameters::getScaling());
183 }
184 else
185 problem->scaleAllData(Parameters::getScaling());
186
187 if(Parameters::getVerbosity() >= 1) {
188 std::cout<< "Performed scaling on target: " << target << std::endl;
189 std::cout<< "Scaling: " << Parameters::getScaling() << std::endl;
190 }
191}

321void Solver::solveAlgorithm(int id) {
322 Algorithm* algorithm;
323
324 if(id == MAIN_ALGORITHM)
325 algorithm = main_algorithm;
326 else if (id < ensemble_algorithms.size())

331 algorithm->setExecutionID(optimization_round);
