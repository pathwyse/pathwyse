209bool LMacyclic::insert(Label& candidate) {
210
211 int node = candidate.getNode();
212 bool direction = candidate.getDirection();
213
214 auto cost_label = bound_labels->getLabel(RES_COST, direction, node);
215 auto res_label = bound_labels->getLabel(RES_CRITICAL, direction, node);
216
217 if(candidate.getObjective() > res_label->getObjective())
218 return false;
219
220 if(candidate.getSnapshot(RES_CRITICAL) > cost_label->getSnapshot(RES_CRITICAL))
221 return false;
222
223 auto cost_label_opp = bound_labels->getLabel(RES_COST, not direction, node);
224 auto res_label_opp = bound_labels->getLabel(RES_CRITICAL, not direction, node);
225
226 int value = candidate.getObjective() + cost_label_opp->getObjective();
227
228 if(value > incumbent)
229 return false;
230
231 if(res_label_opp->isValid())
232 if(not res->isFeasible(candidate.getSnapshot(RES_CRITICAL) + res_label_opp->getSnapshot(RES_CRITICAL)))
233 return false;
234
235 auto & min_consumption = direction ? min_consumption_f[node] : min_consumption_b[node];
236 if(candidate.getSnapshot(RES_CRITICAL) >= min_consumption)
237 return false;
238
239 auto& bucket = direction ? bucket_f[value - offset_f] : bucket_b[value - offset_b];
240 Label* ins = new Label();
241 *ins = candidate;
242 bucket.push_back(ins);
243 direction ? nlabels_fw++ : nlabels_bw++;
244
245 return true;

271bool LMacyclic::earlyUpdate(Label *candidate) {
272 int node = candidate->getNode();
273 bool direction = candidate->getDirection();
274 auto cost_label_opp = bound_labels->getLabel(RES_COST, not direction, node);
275
276 int value = obj->join(candidate->getObjective(), cost_label_opp->getObjective(), node);
277 int res_consumption = candidate->getSnapshot(RES_CRITICAL) + cost_label_opp->getSnapshot(RES_CRITICAL);
278

286

292 res_consumption = candidate->getSnapshot(RES_CRITICAL) + res_label_opp->getSnapshot(RES_CRITICAL);
293 if(value < incumbent and res->isFeasible(res_consumption)){

303//Perform join procedures
304void LMacyclic::join(Label* current) {
305 int node = current->getNode();
306 bool direction = current->getDirection();
307 auto & opposite_closed = direction ? closed_b[node] : closed_f[node];
308
309 double bounding_opp = direction ? 1 - split_ratio : split_ratio;
310 Label* res_label_opp = bound_labels->getLabel(RES_CRITICAL, not direction, node);
311 if(res_label_opp->isValid()) {
312 int res_bound_opp = res_label_opp->getSnapshot(RES_CRITICAL);
313 if(not res->isFeasible(res_bound_opp, node, bounding_opp, direction))
314 return;
315 }
316
317 if(not opposite_closed.empty()) {
318 int cost, consumption;
319 for (auto* closed: opposite_closed) {
320 cost = obj->join(current->getObjective(),closed->getObjective(), node);
321 if(cost > incumbent)
322 break;
323
324 joinComparisons++;
325 consumption = res->join(current->getSnapshot(RES_CRITICAL), closed->getSnapshot(RES_CRITICAL), node);
326 if(res->isFeasible(consumption, node)){
327 incumbent = cost;
328 best_label_pair = std::make_tuple(incumbent, *current, *closed);
329 break;
330 }
331 }
332 }
