93double Preprocessing::getSplit() {
94 double fw_cost = 0, bw_cost = 0;
95 BoundLabels* bound_labels = problem->getBoundLabels();
96 if(bound_labels->getLabel(RES_COST, true, problem->getOrigin())->getObjective() == UNKNOWN)
97 return 0.5;
98
99 for (int node = 0; node < problem->getNumNodes(); node++) {
100
101 if(not problem->isActiveNode(node))
102 continue;
103
104 fw_cost += bound_labels->getLabel(RES_COST, false, node)->getObjective();
105 bw_cost += bound_labels->getLabel(RES_COST, true, node)->getObjective();
106 }
107
108 if(fw_cost > bw_cost)
109 return std::min(1.0, (0.5 * fw_cost / bw_cost));
110 else
111 return 1 - std::min(1.0, (0.5 * bw_cost / fw_cost));
