74 // This function resets the unused bits

104 // resize
105 void resize(std::size_t num_bits, bool value) {
106 size_t old_size = m_bitset.size();
107 size_t required_block = calc_num_blocks(num_bits);
108
109 const uint64_t v = value? std::numeric_limits<uint64_t>::max() : 0;
110
111 if (old_size != required_block) {
112 this->m_bitset.resize(num_bits, v);
113 }
114 if (value && (num_bits > m_num_bits)) {
115 // bistset has been increased
116 size_t extra_bits = bit_index(m_num_bits);
117 if (extra_bits) {
118 assert(old_size >= 1 && old_size <= m_bitset.size());
119 m_bitset[old_size - 1] |= (v << extra_bits);
120 }
121 }
122 m_num_bits = num_bits;

154 template <typename ValueType>
155 std::size_t popcount(ValueType value){
156 std::size_t num = 0u;
157 while (value) {
158 num += count_table<>::table[value & ((1u<<table_width) - 1)];
159 value >>= table_width;
160 }
161 return num;
162 }

164 template <typename T>
165 const unsigned char * object_representation (T* p){
166 return static_cast<const unsigned char *>(static_cast<const void *>(p));
167 }

169 template <typename Iterator>
170 size_t do_count(Iterator first, std::size_t length, bool access_by_bytes){
171 if(!access_by_bytes) {
172 // Access by blocks
173 std::size_t num1 = 0u, num2 = 0u;
174 while (length >= 2u) {
175 num1 += popcount<uint64_t>(*first);
176 ++first;
177 num2 += popcount<uint64_t>(*first);
178 ++first;
179 length -= 2u;
180 }
181
182 if (length > 0u)
183 num1 += popcount<uint64_t>(*first);
184
185 return num1 + num2;
186 } else {
187 // access by bytes
188 if (length > 0u) {
189 const unsigned char* p = object_representation(&*first);
190 length *= sizeof(*first);
191
192 return do_count(p, length, false);
193 }
194 return 0u;
195 }
196 }

198 size_t count(){
199 enum { no_padding = bits_per_uint64 == CHAR_BIT * sizeof(uint64_t) };

272 ios_base::iostate err = ok;
273
274 typename basic_ostream<Ch, Tr>::sentry cerberos(os);
275 if (cerberos) {
276
277 const Ch zero = '0';
278 const Ch one = '1';
279
280 try {
281
282 typedef uint64_t bitset_size_type;
283 typedef basic_streambuf<Ch, Tr> buffer_type;
284
285 buffer_type *buf = os.rdbuf();
286 // careful: os.width() is signed (and can be < 0)
287 const bitset_size_type width = (os.width() <= 0) ? 0 : static_cast<bitset_size_type>(os.width());
288 streamsize npad = (width <= b.size()) ? 0 : width - b.size();
289
290 const Ch fill_char = os.fill();
291 const ios_base::fmtflags adjustfield = os.flags() & ios_base::adjustfield;
292
293 // if needed fill at left; pad is decreased along the way
294 if (adjustfield != ios_base::left) {
295 for (; 0 < npad; --npad)
296 if (Tr::eq_int_type(Tr::eof(), buf->sputc(fill_char))) {
297 err |= ios_base::failbit;
298 break;
299 }
300 }
301
302 if (err == ok) {
303 // output the bitset
304 for (bitset_size_type i = b.size(); 0 < i; --i) {
305 typename buffer_type::int_type
306 ret = buf->sputc(b.get(i - 1) ? one : zero);
307 if (Tr::eq_int_type(Tr::eof(), ret)) {
308 err |= ios_base::failbit;
309 break;
310 }
311 }
312 }
313
314 if (err == ok) {
315 // if needed fill at right
316 for (; 0 < npad; --npad) {
317 if (Tr::eq_int_type(Tr::eof(), buf->sputc(fill_char))) {
318 err |= ios_base::failbit;
319 break;
320 }
321 }
322 }
323
324
325 os.width(0);
326
327 } catch (...) { // see std 27.6.1.1/4
328 bool rethrow = false;
329 try { os.setstate(ios_base::failbit); } catch (...) { rethrow = true; }
330
331 if (rethrow)
332 throw;
333 }
334 }

337 os.setstate(err); // may throw exception
