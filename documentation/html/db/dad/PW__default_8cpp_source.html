187void PWDefault::extend(LabelAdv* candidate) {
188 LabelAdv new_label = LabelAdv();
189 int node = candidate->getNode();
190 std::vector<int> neighbors = problem->getNeighbors(node, candidate->getDirection());
191 bool active;
192
193 for(auto & neigh: neighbors) {
194
195 active = not unreachable_active.empty() and unreachable_active[node].get(neigh);
196
197 if((active and label_manager->isNodeReachable(candidate, neigh)) or
198 (not active and label_manager->isExtensionFeasible(candidate, neigh))){
199
200 //Eliminates k = 2 cycles
201 if(candidate->getPredecessor() and candidate->getPredecessor()->getNode() == neigh)
202 continue;
203
204 //critical res extension check
205 if(bidirectional and not label_manager->isCriticalExtensionFeasible(candidate, neigh))
206 continue;
207
208 //For each reachable node, extend label in that direction
209 label_manager->extendLabel(candidate, &new_label, neigh);
210
211 //Update unreachables
212 label_manager->updateUnreachables(&new_label);
213 if(not unreachable_active.empty())
214 new_label.updateUnreachables(unreachable_active[neigh]);
215
216 if(Parameters::getCollectionLevel()>=1)
217 collector.startTime(candidate->getDirection() ? T_INS_FW : T_INS_BW);
218 label_manager->insert(&new_label);
219 if(Parameters::getCollectionLevel()>=1)
220 collector.stopTime(candidate->getDirection() ? T_INS_FW : T_INS_BW);
221 candidate->getDirection() ? ins_attempts_fw++ : ins_attempts_bw++;
222 }
223 }

226bool PWDefault::checkTermination() {
227
228 bool termination = true;
229 auto bestPath = getBestSolution();
230
231 //Condition 1: Timeout reached
232 if(timeout) {
233 if(bestPath and bestPath->isElementary())
234 updateIncumbent(bestPath->getObjective());
235
236 return termination;
237 }
238
239 //Condition 2: No solution found
240 if(not bestPath) {
241 if(algo_type == ALGO_EXACT)
242 problem->setStatus(PROBLEM_INFEASIBLE);
243 return termination;
244 }
245
246 if(algo_type == ALGO_EXACT)
247 updateLowerBound(bestPath->getObjective());
248
249 //Condition 3: Elementary solution found
250 if(bestPath->isElementary()){
251 updateIncumbent(bestPath->getObjective());
252
253 if(ng != NG_OFF)
254 ng_compliant = true;
255 return termination;
256 }
257
258 //Condition 4: NG solution found (if DSSR is OFF)
259 if(ng == NG_RESTRICTED) {
260 termination = NgRestricted();
261 ng_compliant = termination;
262 }
263
264 //If DSSR is enabled, termination will be false until an elementary solution is found
265 if(dssr != DSSR_OFF and termination) {
266 //NG algorithms are not used anymore
267 ng = NG_OFF;
268 termination = dssr == DSSR_STANDARD ? DssrStandard() : DssrRestricted();
269 }
270
271 //Reset for next iteration
272 if(not termination) {
273 label_manager->resetLM();
274 solutions.clear();
275 }
276
277 return termination;
278}

314bool PWDefault::DssrStandard(){
315 bool isElementary = true;
316
317 Bitset repeated_visits(problem->getNumNodes());
318 auto tour = getBestSolution()->getTour();
319 tour.sort();
320 int current_node = -1;
321
322 for(auto t: tour) {
323 if(t == current_node) {
324 isElementary = false;

385void PWDefault::buildNG(){
386 if(ng_size <= 1) return;
387
388 ng_compliant = true;
389 auto objective = problem->getObj();
390
391 unreachable_ng.resize(problem->getNumNodes(), Bitset(problem->getNumNodes()));
392 std::set<std::pair<int, int>> neighbors;
393
394 //Arc-cost based ordering (should take into account extension instead)
395 int neighborhood_max_size = ng_size - 1; //To account for node i in the set
396 for(int i = 0; i < problem->getNumNodes(); i++) {
397 for(int j = 0; j < problem->getNumNodes(); j++)
398 if(problem->areNeighbors(i, j, true)) {
399 if(neighbors.size() < neighborhood_max_size)
400 neighbors.insert(std::make_pair(objective->extend(0,i,j,true), j));
401 else if(objective->extend(0,i,j,true) < neighbors.rbegin()->first) {
402 neighbors.erase(std::prev(neighbors.end()));
403 neighbors.insert(std::make_pair(objective->extend(0,i,j,true), j));
404 }
405 }
406
407 for(auto & n: neighbors)
408 unreachable_ng[i].set(n.second);
409 unreachable_ng[i].set(i);
410 neighbors.clear();
411 }
412
413 if(ng == NG_STANDARD) {
414 unreachable_active = unreachable_ng;
415 unreachable_max_count = unreachable_active[0].count();
416 }

419bool PWDefault::NgRestricted() {
420 bool isNG = true;
421 Bitset visited_memory(problem->getNumNodes());
422 auto tour = getBestSolution()->getTour();
423 std::list<int> scanned_nodes;
424 int scanned_node;
425
426 for(auto t: tour) {
427 //If visited has memory of seeing a node that is part of the ng-route: cycle on t is forbidden
428 if(visited_memory.get(t)){
429 isNG = false;
430 //Bit to 1 in position t for each node of the closest cycle
431 for(auto it_scan = scanned_nodes.rbegin(); it_scan != scanned_nodes.rend(); it_scan++) {
432 scanned_node = *it_scan;
433 unreachable_active[scanned_node].set(t);
434 if(scanned_node == t) break;
435 }
436 }
437 scanned_nodes.push_back(t);
438 visited_memory &= unreachable_ng[t];
439 visited_memory.set(t);
440 }
441
442 int u_count;
443 for(auto t: tour){
444 u_count = unreachable_active[t].count();
445 if(u_count > unreachable_max_count)
446 unreachable_max_count = u_count;
